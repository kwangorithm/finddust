# OpenAI API로 PM2.5 데이터 자동 리포트 생성
import openai, pandas as pd

def generate_pm25_report(data_path):
    df = pd.read_csv(data_path)
    stats = df.describe().to_string()

    response = openai.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user",
            "content": f"다음 PM2.5 통계를 한국어 논문 결과 문단으로 작성해줘:\n{stats}"}]
    )
    return response.choices[0].message.content

# 실행
report = generate_pm25_report("data/processed/pm25_2023.csv")
print(report)
