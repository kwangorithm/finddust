Codex API 첫 호출

from openai import OpenAI
import os

client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

response = client.chat.completions.create(
    model="gpt-4o",
    messages=[
        {"role": "system",
         "content": "당신은 대기환경 데이터 분석 전문 코딩 어시스턴트입니다."},
        {"role": "user",
         "content": "PM2.5 일별 데이터에서 7일 이동평균을 계산하는 pandas 코드를 작성해줘"}
    ]
)

print(response.choices[0].message.content)
💡 system 프롬프트 활용
system 메시지에 "대기환경 분석 전문가" 역할을 부여하면 더 전문적인 코드를 생성합니다.

⌨️ 실습 2
반복 분석 자동화
🎯 시나리오
서울 25개 구 각각의 PM2.5 CSV 파일을 자동으로 읽어서, 구별 통계 요약 + 트렌드 그래프를 일괄 생성
python — 구별 자동 분석 배치 스크립트
from openai import OpenAI
import pandas as pd, glob

client = OpenAI()
districts = glob.glob("data/processed/district_*.csv")

for filepath in districts:
    df = pd.read_csv(filepath)
    sample = df.head(3).to_string()  # 데이터 구조 전달

    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[{
            "role": "user",
            "content": f"다음 데이터 구조로 연도별 트렌드 분석 코드 작성:\n{sample}"
        }]
    )
    # 생성된 코드를 파일로 저장
    code = response.choices[0].message.content
    Path(f"src/auto_{Path(filepath).stem}.py").write_text(code)

📈 실습 3
PM2.5 시계열 분석
python — 계절 분해 (STL Decomposition)
from statsmodels.tsa.seasonal import STL
import matplotlib.pyplot as plt

# STL 계절 분해: 추세 / 계절성 / 잔차 분리
stl = STL(df.set_index('datetime')['pm25'],
          period=12, robust=True)
result = stl.fit()

fig = result.plot()
fig.suptitle('서울 PM2.5 STL 계절 분해 (2015–2023)')
fig.savefig('figures/fig_stl.tiff', dpi=300)
STL이 분리하는 것
Trend — 장기 감소 추세
Seasonal — 계절 패턴 (겨울↑ 여름↓)
Residual — 황사 등 이벤트
Claude Code 프롬프트
"STL 분해 결과에서 계절 진폭이 연도별로 변하는지 확인하는 코드 추가해줘"

🔍 실습 4
이상값 자동 탐지 (황사 이벤트)
python — IQR 기반 이상값 탐지 & 표시
import numpy as np

# IQR 방법으로 이상값 탐지
Q1 = df['pm25'].quantile(0.25)
Q3 = df['pm25'].quantile(0.75)
IQR = Q3 - Q1
outliers = df[df['pm25'] > Q3 + 1.5 * IQR]

print(f"이상값(황사 의심) 날짜: {len(outliers)}일")
print(outliers[['datetime', 'pm25']].head(10))

# 시각화에 표시
ax.scatter(outliers['datetime'], outliers['pm25'],
           color='red', s=20, zorder=5, label='이상값')
💡 AI 활용 팁
탐지된 이상값 날짜를 Claude Code에 붙여넣고 "이 날짜들의 공통점을 기상청 황사 기록과 비교하는 방법 알려줘" 라고 물어보세요.

논문 품질 Figure 고급 설정
python — 저널 제출 기준 맞춤 Figure
import matplotlib.pyplot as plt
import matplotlib as mpl

# 저널 공통 설정
mpl.rcParams.update({
    'font.size': 8, 'axes.labelsize': 9,
    'font.family': 'NanumGothic',
    'axes.linewidth': 0.8, 'xtick.major.width': 0.8,
})

# 단컬럼(88mm) / 이중컬럼(180mm) 설정
fig, axes = plt.subplots(1, 2,
    figsize=(7.09, 3.0))  # 180mm × 76mm

# 저장: 300dpi TIFF (대부분 저널 요구사항)
fig.savefig('figures/fig2_analysis.tiff',
            dpi=300, format='tiff',
            bbox_inches='tight', pil_kwargs={'compression':'tiff_lzw'})
💡 저널별 Figure 규정
투고 전 "저널명 + figure guidelines" 검색 필수. 일반적으로 300dpi, TIFF/EPS, 컬럼 너비 준수.

import seaborn as sns

fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(7.09, 3.0))

# 계절별 박스플롯
df['season'] = df['month'].map({
    12:'겨울',1:'겨울',2:'겨울',
    3:'봄',4:'봄',5:'봄',
    6:'여름',7:'여름',8:'여름',
    9:'가을',10:'가을',11:'가을'})
sns.boxplot(data=df, x='season', y='pm25', ax=ax1,
            order=['봄','여름','가을','겨울'])

# 월별 연도별 히트맵
pivot = df.pivot_table('pm25', 'year', 'month')
sns.heatmap(pivot, ax=ax2, cmap='YlOrRd', cbar_kws={'label':'μg/m³'})
💬 Claude Code 프롬프트
"위 Figure를 저널 투고 기준에 맞게 다듬어줘. 폰트 크기 8pt, 박스플롯에 평균값 다이아몬드 표시 추가, 패널 레이블 (a) (b) 붙이기"
